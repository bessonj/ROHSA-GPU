2  *
3  * lbfgsb_wrapper.c, mex gatewate interface
4  * You can call this directly from Matlab,
5  * but I strongly suggest using lbfgsb.m, since many
6  * variables are modified in-place, and lbfgsb.m will
7  * handle them correctly. It also shows you how to use
8  * this routine
9  * (so just modify lbfgsb.m to suit your own taste)
10  *
11  * Written by Stephen Becker, Feb 14 2012
12  * Major update Feb 21 2015
13  *
14  * This re-distributes a C version of lbfgsb 3.0
15  * (free; see http://users.eecs.northwestern.edu/~nocedal/lbfgsb.html)
16  *
17  * There are also versions 2.1 and 2.4 of the lbfgsb library
18  * For v 2.1, Peter Carbonetto's mex interface works; see
19  * http://www.mathworks.com/matlabcentral/fileexchange/15061-matlab-interface-for-l-bfgs-b
20  * and also
21  * http://www.cs.ubc.ca/~pcarbo/lbfgsb-for-matlab.html
22  *
23  * For v 2.4 (I don't know where you can find this version though),
24  * use the mex files from here:
25  * http://www.cs.toronto.edu/~liam/software.shtml
26  *
27  *
28  * You should use version 3.0! It's better. The old versions are basically
29  * 15 years old; changes in version 3.0 are fundamental (see the 2011 "remark" paper
30  * listed at Nocedal's website)
31  *
32  * Also, this version has converted everything to C, so there is no mixed
33  * C and fortran, and compilation is much easier. In the conversion,
34  * I changed a few conventions, and some of these are marked with my
35  * initials SRB
36  * -- Stephen Becker, Feb 2015. stephen.becker@colorado.edu
37  *
38  *
39  *
40  * Inputs (in order):
41  * m # of memory vectors to keep
42  * x initial guess or current point, also used to determine the size of the problem (Nx1)
43  * l list of upper bounds (Nx1)
44  * u list of lower bounds (Nx1)
45  * nbd list of which bounds are active (Nx1):
46  * 0=neither u nor l, 1 = only l, 2 = both u and l, 3 = only u
47  * f value of function (1x1)
48  * g gradient of function (Nx1)
49  *
50  * factr stopping crit: 1e+12 for low accuracy, 1e7 for moderate, 1e1 for high accuracy
51  * (will be multiplied by machine precision)
52  * pgtol stopping crit for infinity norm of projected gradient
53  * wa work space array (double)
54  * iwa work space array (int)
55  * task string SRB UPDATE: now an integer
56  * iprint how verbose the fortran program should be
57  * csave some iformation (string, length 60) SRB UPDATE: now an integer
58  * lsave[4] some information (logicals)
59  * isave[44] some information (integers)
60  * dsave[29] some information (doubles)
61  *
62  * Outputs:
63  * f, task, csave, lsave, isave, dsave
64  *
65  * Warning: the following variables are modified in-place
66  * x, g, wa, iwa
67  *
68  *=================================================================*/

110 mxLogical isInt( const mxArray *pm ) {
111  /* We typedef "integer" to be "long int", and this is not
112  * constant across different computers.
113  *
114  * CHAR_BIT is from limits.h
115  * If using gcc, you can run `gcc -dM -E - < /dev/null | grep CHAR_BIT`
116  * and it should define the symbol __CHAR_BIT__, so this is another way.
117  *
118  * This will match the typdef "integer" which is what
119  * the lbfgsb codes uses for integers.
120  * */
121 
122  /* debugPrintf("Sizeof(int) is %d\n", sizeof(int) ); */
123  switch ( CHAR_BIT * sizeof(integer) ) {
124  case 16 :
125  return mxIsInt16(pm);
126  case 32:
127  return mxIsInt32(pm);
128  case 64:
129  return mxIsInt64(pm);
130  default:
131  mexErrMsgTxt("You have a weird computer that I don't know how to support");
132  return false;
133  }
134 }

137 /* Main mex gateway routine */
138 void mexFunction( int nlhs, mxArray *plhs[], int nrhs, const mxArray*prhs[] ) {
139 
140  integer iprint = (integer)1;
141  integer task=(integer)START, csave=(integer)1;
142  integer iterations = 0;
143  integer total_iterations = 0;
144 
145  int iterMax = 100;
146  int total_iterMax = 200;
147 
148 
149  integer n, m, *nbd=NULL, *iwa=NULL;
150  double f=0, factr, pgtol, *x, *l, *u, *g, *wa=NULL;
151  int i;
152  mxLogical FREE_nbd=false;
153 
154  int ndim = 2; /* for lcc compiler, must declare these here, not later ... */
155  mwSize dims[2] = { LENGTH_ISAVE, 1 };
156 
157  logical lsave[LENGTH_LSAVE];
158  integer isave[LENGTH_ISAVE];
159  double dsave[LENGTH_DSAVE];
160 
161  double *nbd_dbl=NULL;
162  long long *nbd_long=NULL;
163 
164  mxArray *LHS[2];
165  mxArray *RHS[3];
166  double *tempX, *tempG, *tempIter;
167 
168  /* Parse inputs. Quite boring */
169 
170  if (nrhs < 5 ) mexErrMsgTxt("Needs at least 5 input arguments");
171  m = (int)*mxGetPr( prhs[N_m] );
172  n = (integer)mxGetM( prhs[N_x] );
173  if ( mxGetN(prhs[N_x]) != 1 ) mexErrMsgTxt("x must be a column vector");
174  if ( mxGetM(prhs[N_l]) != n ) mexErrMsgTxt("l must have same size as x");
175  if ( mxGetM(prhs[N_u]) != n ) mexErrMsgTxt("u must have same size as x");
176  if ( mxGetM(prhs[N_nbd]) != n ) mexErrMsgTxt("nbd must have same size as x");
177 
178 
179  if (nlhs < 2 ) mexErrMsgTxt("Should have 2 or 3 output arguments");
180  if (!mxIsDouble(prhs[N_x]))
181  mexErrMsgTxt("x should be of type double!\n");
182  plhs[1] = mxDuplicateArray( prhs[N_x] );
183  x = mxGetPr( plhs[1] );
184 
185 
186  l = mxGetPr( prhs[N_l] );
187  u = mxGetPr( prhs[N_u] );
188  if ( isInt( prhs[N_nbd] ) ) {
189  nbd = (integer *)mxGetData( prhs[N_nbd] );
190  } else {
191  debugPrintf("Converting nbd array to integers\n" );
192  if (!mxIsDouble(prhs[N_nbd])){
193  if (mxIsInt64(prhs[N_nbd])){
194  nbd_long = mxGetData( prhs[N_nbd] );
195  nbd = (integer *)mxMalloc( n * sizeof(integer) );
196  assert( nbd != NULL );
197  FREE_nbd = true;
198  /* convert nbd_dbl (in double format) to integers */
199  for (i=0;i<n;i++)
200  nbd[i] = (integer)nbd_long[i];
201  } else {
202  debugPrintf("Sizeof(int) is %d bits, sizeof(integer) is %d bits\n",
203  CHAR_BIT*sizeof(int),CHAR_BIT*sizeof(integer) );
204  /* integer is aliased to 'long int' and should be at least
205  * 32 bits. 'long long' should be at least 64 bits.
206  * On 64-bit Windows, it seems 'long int' is exactly 32 bits,
207  * while on 64-bit linux and Mac, it is 67 bits */
208  debugPrintf("Nbd is of type %s\n", mxGetClassName( prhs[N_nbd] ) );
209  mexErrMsgTxt("Nbd array not doubles or type int64!\n");
210  }
211  } else {
212  nbd_dbl = mxGetPr( prhs[N_nbd] );
213  nbd = (integer *)mxMalloc( n * sizeof(integer) );
214  assert( nbd != NULL );
215  FREE_nbd = true;
216  /* convert nbd_dbl (in double format) to integers */
217  for (i=0;i<n;i++)
218  nbd[i] = (integer)nbd_dbl[i];
219  }
220  }
221 
222 
223  /* some scalar parameters */
224  if ( nrhs < N_factr+1 )
225  factr = 1.0e7;
226  else if (mxGetNumberOfElements( prhs[N_factr] )!=1)
227  factr = 1.0e7;
228  else {
229  factr = (double)mxGetScalar( prhs[N_factr] );
230  if (factr < 0 )
231  mexErrMsgTxt("factr must be >= 0\n");
232  }
233 
234  if ( nrhs < N_pgtol+1 )
235  pgtol = 1.0e-5;
236  else if (mxGetNumberOfElements( prhs[N_pgtol] )!=1)
237  pgtol = 1.0e-5;
238  else {
239  pgtol = (double)mxGetScalar( prhs[N_pgtol] );
240  if (pgtol < 0)
241  mexErrMsgTxt("pgtol must be >= 0\n");
242  }
243  if ( nrhs < N_iprint+1 ) {
244  iprint = (integer)1;
245  } else if (mxGetNumberOfElements( prhs[N_iprint] )!=1) {
246  iprint = (integer)1;
247  } else {
248  iprint = (integer)mxGetScalar( prhs[N_iprint] );
249  }
250 
251  if ( nrhs >= N_iterMax+1 )
252  iterMax = (int)mxGetScalar( prhs[N_iterMax] );
253  if ( nrhs >= N_total_iterMax+1 )
254  total_iterMax = (int)mxGetScalar( prhs[N_total_iterMax] );
255 
256  /* allocate memory for arrays */
257  g = (double *)mxMalloc( n * sizeof(double) );
258  assert( g != NULL );
259  wa = (double *)mxMalloc( (2*m*n + 5*n + 11*m*m + 8*m ) * sizeof(double) );
260  assert( wa != NULL );
261  iwa = (integer *)mxMalloc( (3*n)*sizeof(integer) );
262  assert( iwa != NULL );
263 
264 
265 
266  /* -- Finally, done with parsing inputs. Now, call lbfgsb fortran routine */
267 
268  /* Be careful! This modifies many variables in-place!
269  * Basically, anything without a '&' before it will be changed in the Matlab
270  * workspace */
271 
272  if ( nrhs < N_fcn - 1 )
273  mexErrMsgTxt("For this f(x) feature, need more input aguments\n");
274  RHS[0] = mxDuplicateArray( prhs[N_fcn] );
275  RHS[1] = mxCreateDoubleMatrix(n,1,mxREAL);
276  RHS[2] = mxCreateDoubleScalar( 0.0 ); /* The iterations counter */
277  tempX = (double*)mxGetPr( RHS[1] );
278  if (!mxIsDouble(RHS[2]))
279  mexErrMsgTxt("Error trying to create RHS[2]\n");
280  tempIter = (double*)mxGetPr( RHS[2] );
281 
282  while ( (iterations < iterMax) && (total_iterations < total_iterMax ) ){
283  total_iterations++;
284 
285  setulb(&n,&m,x,l,u,nbd,&f,g,&factr,&pgtol,wa,iwa,&task,&iprint,
286  &csave,lsave,isave,dsave); /* (ftnlen) TASK_LEN, (ftnlen) CSAVE_LEN); */
287 
288 
289  if ( IS_FG(task) ) {
290 
291  /* copy data from x to RHS[1] or just set pointer with mxSetPr */
292  for (i=0;i<n;i++)
293  tempX[i] = x[i];
294  /*Try being bold: */
295  /*mxSetPr( RHS[1], x ); */
296 
297  *tempIter = (double)iterations;
298  mexCallMATLAB(2,LHS,3,RHS,"feval");
299  f = mxGetScalar( LHS[0] );
300  if (mxGetM(LHS[1]) != n )
301  mexErrMsgTxt("Error with [f,g]=fcn(x) : g wrong size\n");
302  if (mxGetN(LHS[1]) != 1 )
303  mexErrMsgTxt("Error with [f,g]=fcn(x) : g wrong size (should be column vector)\n");
304 
305  /* could use memcpy, or just do it by hand... */
306  if (!mxIsDouble(LHS[1]))
307  mexErrMsgTxt("[f,g]=fcn(x) did not return g as type double\n");
308  tempG = mxGetPr( LHS[1] );
309  for (i=0;i<n;i++)
310  g[i] = tempG[i];
311  /* Or, be a bit bolder: */
312  /*g = tempG; // Hmm, crashed */
313 
314  continue;
315  }
316  if ( task==NEW_X ) {
317  iterations++;
318  continue;
319  } else
320  break;
321 
322  }
323 
324  mxDestroyArray( LHS[0] );
325  mxDestroyArray( LHS[1] );
326  mxDestroyArray( RHS[0] );
327  mxDestroyArray( RHS[1] );
328 
329 
330 
331  plhs[0] = mxCreateDoubleScalar( f );
332  if ( nlhs >= 3 )
333  plhs[2] = mxCreateDoubleScalar( task );
334  if ( nlhs >= 4 )
335  plhs[3] = mxCreateDoubleScalar( iterations );
336  if ( nlhs >= 5 )
337  plhs[4] = mxCreateDoubleScalar( total_iterations );
338  if ( nlhs >= 6 )
339  mexErrMsgTxt("Did not expect more than 5 outputs\n");
340 
341  if (FREE_nbd)
342  mxFree(nbd);
343  mxFree(g);
344  mxFree(wa);
345  mxFree(iwa);
346 
347  return;
348 }

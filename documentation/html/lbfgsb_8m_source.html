1 function [x,f,info] = lbfgsb( fcn, l, u, opts )
2 % x = lbfgsb( fcn, l, u )
3 % uses the lbfgsb v.3.0 library (fortran files must be installed;
4 % see compile_mex.m ) which is the L-BFGS-B algorithm.
5 % The algorithm is similar to the L-BFGS quasi-Newton algorithm,
6 % but also handles bound constraints via an active-set type iteration.
7 % This version is based on the modified C code L-BFGS-B-C, and so has
8 % a slightly different calling syntax than previous versions.
9 %
10 % The minimization problem that it solves is:
11 % min_x f(x) subject to l <= x <= u
12 %
13 % 'fcn' is a function handle that accepts an input, 'x',
14 % and returns two outputs, 'f' (function value), and 'g' (function gradient).
15 %
16 % 'l' and 'u' are column-vectors of constraints. Set their values to Inf
17 % if you want to ignore them. (You can set some values to Inf, but keep
18 % others enforced).
19 %
20 % The full format of the function is:
21 % [x,f,info] = lbfgsb( fcn, l, u, opts )
22 % where the output 'f' has the value of the function f at the final iterate
23 % and 'info' is a structure with useful information
24 % (self-explanatory, except for info.err. The first column of info.err
25 % is the history of the function values f, and the second column
26 % is the history of norm( gradient, Inf ). )
27 %
28 % The 'opts' structure allows you to pass further options.
29 % Possible field name values:
30 %
31 % opts.x0 The starting value (default: all zeros)
32 % opts.m Number of limited-memory vectors to use in the algorithm
33 % Try 3 <= m <= 20. (default: 5 )
34 % opts.factr Tolerance setting (see this source code for more info)
35 % (default: 1e7 ). This is later multiplied by machine epsilon
36 % opts.pgtol Another tolerance setting, relating to norm(gradient,Inf)
37 % (default: 1e-5)
38 % opts.maxIts How many iterations to allow (default: 100)
39 % opts.maxTotalIts How many iterations to allow, including linesearch iterations
40 % (default: 5000)
41 % opts.printEvery How often to display information (default: 1)
42 % opts.errFcn A function handle (or cell array of several function handles)
43 % that computes whatever you want. The output will be printed
44 % to the screen every 'printEvery' iterations. (default: [] )
45 % Results saved in columns 3 and higher of info.err variable
46 %
47 % Stephen Becker, srbecker@alumni.caltech.edu
48 % Feb 14, 2012
49 % Updated Feb 21 2015, Stephen Becker, stephen.becker@colorado.edu

103 % 'nbd' is 0 if no bounds, 1 if lower bound only,
104 % 2 if both upper and lower bounds, and 3 if upper bound only.
105 % This .m file assumes l=-Inf and u=+Inf imply that there are no constraints.
106 % So, convert this to the fortran convention:
107 nbd = isfinite(l) + isfinite(u) + 2*isinf(l).*isfinite(u);
108 if ispc

115 % Some scalar settings, "factr" and "pgtol"
116 % Their descriptions, from the fortran file:
117 
118 % factr is a DOUBLE PRECISION variable that must be set by the user.
119 % It is a tolerance in the termination test for the algorithm.
120 % The iteration will stop when
121 %
122 % (f^k - f^{k+1})/max{|f^k|,|f^{k+1}|,1} <= factr*epsmch
123 %
124 % where epsmch is the machine precision which is automatically
125 % generated by the code. Typical values for factr on a computer
126 % with 15 digits of accuracy in double precision are:
127 % factr=1.d+12 for low accuracy;
128 % 1.d+7 for moderate accuracy;
129 % 1.d+1 for extremely high accuracy.
130 % The user can suppress this termination test by setting factr=0.
131 factr = setOpts( 'factr', 1e7, 0 );
132 
133 % pgtol is a double precision variable.
134 % On entry pgtol >= 0 is specified by the user. The iteration
135 % will stop when
136 %
137 % max{|proj g_i | i = 1, ..., n} <= pgtol
138 %
139 % where pg_i is the ith component of the projected gradient.
140 % The user can suppress this termination test by setting pgtol=0.
141 pgtol = setOpts( 'pgtol', 1e-5, 0 ); % may crash if < 0
142

177 function [f,g] = fcn_wrapper( callF, errFcn, maxIts, printEvery, x, varargin )
178 persistent k history
179 if isempty(k), k = 1; end
180 if nargin==4
181  % reset persistent variables and return information
182  if ~isempty(history) && ~isempty(k)
183  if printEvery > 0
184  printFcn(k,history);
185  end
186  f = history(1:k,:);
187  end
188  history = [];
189  k = [];
190  return;
191 end
192 if isempty( history )
193  width = 0;
194  if iscell( errFcn ), width = length(errFcn);
195  elseif ~isempty(errFcn), width = 1; end
196  width = width + 2; % include fcn and norm(grad) as well
197  history = zeros( maxIts, width );
198 end
199 
200 % Find function value and gradient:
201 [f,g] = callF(x);
202 
203 if nargin > 5
204  outerIter = varargin{1}+1;
205 
206  history(outerIter,1) = f;
207  history(outerIter,2) = norm(g,Inf); % g is not projected
208  if isa( errFcn, 'function_handle' )
209  history(outerIter,3) = errFcn(x);
210  elseif iscell( errFcn )
211  for j = 1:length(errFcn)
212  history(outer_count,j+2) = errFcn{j}(x);
213  end
214  end
215 
216  if outerIter > k
217  % Display info from *previous* input
218  % Since this may be called several times before outerIter
219  % is actually updated
220 % fprintf('At iterate %5d, f(x)= %.2e, ||grad||_infty = %.2e [MATLAB]\n',...
221 % k,history(k,1),history(k,2) );
222  if (printEvery > 0) && ~isinf(printEvery) && ~mod(k,printEvery)
223  printFcn(k,history);
224  end
225  k = outerIter;
226  end
227 
228 
229 end
230 
231 end

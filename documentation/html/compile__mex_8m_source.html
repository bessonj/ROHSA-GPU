48 disp('=== lbfgsb "driver1" test problem (Rosenbrock, 25 dimensions) === ');
49 % Here's the test problem included with lbfgsb called 'driver1'
50 % (It's a version of the Rosenbrock test function)
51 
52 n = 25;
53 
54 l = ones(n,1); u = l;
55 odd = 1:2:n;
56 even= 2:2:n;
57 l(odd) = 1.0;
58 u(odd) = 1.0e2;
59 l(even)= -1.0e2;
60 u(even)= 1.0e2;
61 
62 opts = struct( 'x0', 3*ones(n,1) );
63 opts.printEvery = 2; % controls how often we print output from .m file wrapper
64 opts.m = 5;
65 %opts.maxIts = 10;
66 opts.errFcn = @(x) norm(x-1); % for now just an arbitrary fcn
67 % opts.verbose = -1; % default is -1, i.e., no output from mex
68 
69 [x,f,info] = lbfgsb( @driver1, l, u, opts );
70 
71 % The true objective value is 0.

73  disp('Success!');
74  semilogy( abs(info.err(:,1)-f),'o-' );
75  xlabel('iteration');
76  ylabel('error in objective function');

85 %% another test function, the 2D Rosenbrock function
86 disp('=== Rosenbrock test function, 2D === ');
87 n = 2;
88 
89 fxy = @(x,y) 100*( y-x.^2).^2 + (1-x ).^2 ;
90 f = @(x) fxy( x(1,:), x(2,:) );
91 gxy = @(x,y) [100*(4*x.^3-4*x.*y)+2*x-2; 100*(2*y-2*x.^2)];
92 g = @(x) gxy( x(1,:), x(2,:) );
93 
94 % There are no constraints
95 l = -inf(n,1);
96 u = inf(n,1);
97 
98 opts = struct( 'x0', [-1.9;2] );
99 opts.printEvery = 1;
100 opts.m = 5;
101 
102 % Here's an example of using an error function. For Rosenbrock,
103 % we know the true solution, so we can measure the error at every
104 % iteration:
105 trueSoln = [1;1];
106 % "errFcn" will be printed to the screen
107 opts.errFcn = @(x) norm(x-trueSoln)/max(norm(trueSoln),1);
108 % "outputFcn" will save values in the "info" output
109 opts.outputFcn = opts.errFcn;
110 
111 % Ask for very high accuracy
112 opts.pgtol = 1e-10;
113 opts.factr = 1e3;
114 
115 % The {f,g} is another way to call it
116 [x,f,info] = lbfgsb( {f,g} , l, u, opts );
117 
118 if abs(f) < 1e-8
119  disp('Success!');
120 % since we included opts.outputFcn, the info.err now has 3 columns.
121 % The first 2 columns are the same as before; the 3rd column
122 % is the output of our outputFcn
123 semilogy( info.err(:,3)-f,'o-' ); xlabel('iteration'); ylabel('relative error in iterate function');
124 else
125  disp('Something didn''t work right :-( ');
126 end

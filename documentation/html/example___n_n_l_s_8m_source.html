1 %% Non-negative least-squares (NNLS) using L-BFBS-B
2 %
3 % Non-negative least-squares solves the following problem:
4 % $$ \min_x \|Ax \textrm{--} b\|^2_2 \quad\textrm{such that}\quad x \ge 0 $$
5 %
6 % The matrix 'A' may have more columns than rows (the 'underdetermined' case),
7 % or more rows than columns (the 'overdetermined' case), or the same
8 % number of rows and columns. Some solvers, such as the PQN
9 % method described in "Tackling Box-Constrained Optimization via a new
10 % Projected Quasi-Newton Approach" by Dongmin Kim, Suvrit Sra, and
11 % Inderjit Dhillon (http://www.cs.utexas.edu/users/inderjit/public_papers/pqnj_sisc10.pdf),
12 % only work for the overdetermined case.
13 %
14 % To quote from that paper,
15 % "Not surprisingly, some constrained
16 % optimization methods have also been applied to solve NNLS. It is interesting
17 % to note that for large scale problems these specialized algorithms are outperformed
18 % by modern methods such as TRON, LBFGS-B, or the methods of this paper. Curiously
19 % this fact has not yet been widely adopted by the wider research community
20 % (footnote: This could be because Matlab continues to ship the antiquated
21 % lsqnonneg function, which is an implementation of the original NNLS algorithm of
22 % Lawson and Hanson 1974 )."
23 %
24 % The Kim/Sra/Dhillon paper compares the following algorithms:
25 %
26 % Fast NNLS, by Rasmus Bro. Available at:
27 % http://www.mathworks.com/matlabcentral/fileexchange/3388-nnls-and-constrained-regression
28 %
29 % mtron, mex wrapper by Christoph Ortner, available at:
30 % http://www.mathworks.com/matlabcentral/fileexchange/14848-mtron
31 % Based on the fortran tron algorithm by Chih-Jen Lin and Jorge More,
32 % "Newton's method for large bound-constrained optimization problems",
33 % SIAM Journal on Optimization, 9(4), pp. 1100-1127, 1999.
34 % http://www-unix.mcs.anl.gov/~more/tron/
35 %
36 % L-BFGS-B.
37 % R. Byrd, P. Lu, J. Nocedal, and C. Zhu, "A Limited Memory Algorithm
38 % for Bound Constrained Optimization", SIAM Journal on Scientific Computing, 16
39 % (1995), pp. 1190--1208.
40 %
41 %
42 % For a published version of this demo, see
43 % http://www.mathworks.com/examples/matlab/4139-non-negative-least-squares-nnls-using-l-bfbs-b
44 %% This demo
45 % Here, we use the mex wrapper for L-BFGS-B v3.0, which is a significantly
46 % improved version of L-BFGS-B from v2.1. We show how to use
47 % the software and the fminunc_wrapper helper file.
48 %
49 % It also compares to some NNLS implementations availabe on the matworks
50 % file exchange. In addition to Fast NNLS (FNNLS), mtron, and LBFGS,
51 % we compare with the following algorithms, all written by Uriel Roque
52 % and based on: Portugal, Judice and Vicente,
53 % "A comparison of block pivoting and interior point algorithms for
54 % linear least squares problems with nonnegative variables",
55 % Mathematics of Computation, 63(1994), pp. 625-643
56 %
57 % activeset.m This is pretty fast for medium-scale and smaller problems
58 % http://www.mathworks.com/matlabcentral/fileexchange/10908-active-set-algorithm
59 %
60 % blocknnls.m Similar to activeset.m in performance
61 % http://www.mathworks.com/matlabcentral/fileexchange/8157-nnls/content/blocknnls.m
62 %
63 % newton.m Very slow for large problems
64 % http://www.mathworks.com/matlabcentral/fileexchange/10953-newton-s-algorithm-for-nnls
65 %
66 % pcnnls.m (predictor-corrector method) Very slow for large problems
67 % http://www.mathworks.com/matlabcentral/fileexchange/8150-predictor-corrector-algorithm
68 %
69 % The most interesting tests use large matrices. For small matrices, tests
70 % are pointless, because any of the methods are suitable.
71 %% Setup a problem
72 
73 % The best codes handle N = 20,000 as long as the matrix is very sparse.
74 % N = 3000; M = 4000; % Large scale. Things start to get interesting
75 N = 1000; M = 1500; % at this size, some algo take a long time!
76 % N = 100; M = 150; % at this size, all algorithms take < 14 seconds
77 A = randn(M,N);
78 b = randn(M,1);
79 
80 fcn = @(x) norm( A*x - b)^2;
81 % here are two equivalent ways to make the gradient. grad2 is sometimes faster
82 grad1 = @(x) 2*A'*(A*x-b);
83 AtA = A'*A; Ab = A'*b;
84 grad2 = @(x) 2*( AtA*x - Ab );
85 
86 grad = grad2;
87 
88 
89 x = [];
90 time = [];
91 
92 %% Solve NNLS with L-BFGS-B
93 
94 l = zeros(N,1); % lower bound
95 u = inf(N,1); % there is no upper bound
96 tstart=tic;
97 fun = @(x)fminunc_wrapper( x, fcn, grad);
98 % Request very high accuracy for this test:
99 opts = struct( 'factr', 1e4, 'pgtol', 1e-8, 'm', 10);
100 opts.printEvery = 5;
101 if N > 10000
102  opts.m = 50;
103 end
104 % Run the algorithm:
105 [xk, ~, info] = lbfgsb(fun, l, u, opts );
106 t=toc(tstart)
107 % Record results
108 x.lbfgsb = xk;
109 time.lbfgsb = t;
110 
111 
112 %% Solve with TRON, via MTRON interface
113 
114 % Only run this if you have mtron installed and it is in the path
115 if exist( 'itron.m', 'file' )
116 
117  x0 = zeros(N,1);
118  xl = zeros(N,1);
119  xu = +1e300*ones(N,1);
120  fmin = -1e300;
121  H = sparse(AtA/2); % will crash if not a sparse matrix
122  tstart=tic;
123  hess = @(x) H;
124  fun = @(x)fminunc_wrapper( x, fcn, grad, hess );
125  [xk, fval, exitflag, output] = itron(fun, x0, xl, xu, fmin );
126  t=toc(tstart)
127  x.tron = xk;
128  time.tron = t;
129 end
130 
131 %% Active set. Fast on medium problems
132 if exist( 'activeset.m', 'file' )
133  tstart=tic;
134  [xk,y] = activeset(A,b);
135  t=toc(tstart)
136  x.activeset = xk;
137  time.activeset = t;
138 end
139 %% Block pivoting. Fast on medium problems
140 if exist( 'blocknnls.m', 'file' )
141  tstart=tic;
142  [xk] = blocknnls(A,b, 'fixed');
143  t=toc(tstart)
144  x.blockPivot = xk;
145  time.blockPivot = t;
146 end
147 %% Newton. Slow!
148 if exist( 'newton.m', 'file' ) && N < 500
149  tstart=tic;
150  [xk,y] = newton(A,b, ones(N,1), 100); % can't have 0 starting vector
151  t=toc(tstart)
152  x.newton = xk;
153  time.newton = t;
154 else
155  fprintf('Skipping Newton method because we can''t find it, or it is too slow\n');
156 end
157 %% Predictor-Corrector. Can be very slow
158 if exist( 'pcnnls.m', 'file' ) && N < 500
159  tstart=tic;
160  [xk,y,nits] = pcnnls(A,b,ones(N,1), 3000);
161  t=toc(tstart)
162  x.predCorr = xk;
163  time.predCorr = t;
164 else
165  fprintf('Skipping predCorr method because we can''t find it, or it is too slow\n');
166 end
167 %% Run Matlab's default (Lawson and Hanson) Very slow on large problems
168 tstart=tic;
169 xk = lsqnonneg(A,b);
170 t=toc(tstart)
171 x.lsqnonneg = xk;
172 time.lsqnonneg = t;
173 %% Fast NNLS, modification of Lawson and Hanson. Much better for large problems
174 if exist( 'fnnls.m', 'file' )
175  tstart=tic;
176  [xk] = fnnls(A'*A,A'*b);
177  t=toc(tstart)
178  x.fnnls = xk;
179  time.fnnls = t;
180 end
181 %% PQN-LBFGS and PQN-BB algorithms of Kim/Sra/Dhillon. Very fast.
182 if exist( 'solnls.m', 'file' )
183  opt = solopt;
184  opt.maxtime = 2000;
185  opt.verbose = 0;
186  tstart=tic;
187  % run their 'BB' variant
188  opt.algo = 'BB';
189  out = solnls( A, b, zeros(N,1), opt );
190  t=toc(tstart)
191  x.PQN_BB = out.x;
192  time.PQN_BB = t;
193 
194  % and run their 'PLB' variant (their 'PQN' variant is much slower)
195  % which uses L-BFGS (not to be confused with L-BFGS-B)
196  opt.algo = 'PLB';
197 
198  tstart=tic;
199  out = solnls( A, b, zeros(N,1), opt );
200  t=toc(tstart)
201 
202  x.PQN = out.x;
203  time.PQN = t;
204 
205 end
206 %% Results
207 % Find the best answer, and use that as the reference.
208 fMin = Inf;
209 for f=fieldnames(x)',
210  if fcn(x.(f{1})) < fMin,
211  fMin = fcn(x.(f{1}));
212  best = f{1};
213  end
214 end
215 xReference = x.(best);
216 errFcn = @(x) norm(x-xReference)/norm(xReference);
217 
218 % Print out info. Verify that the solution is indeed non-negative (hence the
219 % min(x) information), and the objective function, and the error
220 % against the reference solution. Also display the time.
221 fprintf('== Size of problem is %d x %d == \n', M, N );
222 for f=fieldnames(x)',
223  fprintf('%10s: obj is %7.2f, min(x) is %7.1d, err is %.2e, time is %6.3f s\n', ...
224  f{1}, fcn(x.(f{1})), min(x.(f{1})), errFcn(x.(f{1})), time.(f{1}) );
225 end

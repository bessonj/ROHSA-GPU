129 /* First decision: use the included BLAS code
130  * (un-optimized version taken from Netlib;
131  * this is the "reference" implementation, or
132  * "Ref" for short; this is your best option
133  * since these routines should not be a bottleneck
134  * in your computation);
135  * or, use your own BLAS library, such
136  * as Intel MKL, ATLAS, GotoBLAS, etc.
137  * (For example, you could use libmwblas
138  * or libmwblas_compat32, which are included
139  * with Mathworks and usually based on Intel MKL).
140  *
141  * The reason not to use your own BLAS library
142  * is that (1) it may be faster but it won't be
143  * a bottleneck, and (2) some BLAS libraries
144  * use 32-bit/4byte integers, and others use 64-bit/
145  * 8byte integers, and you pass by reference,
146  * so if you get it wrong, you crash.
147  * In Matlab, useing -lmwblascompat32
148  * uses names like ddot32
149  *
150  * In short, use the Ref version unless you feel
151  * lucky.
152  * */
153 #if !defined( _USE_OPTIMIZED_BLAS )
154 /* make alias to our reference implementation */
155 #define daxpy FORTRAN_WRAPPER(daxpyRef)
156 #define dcopy FORTRAN_WRAPPER(dcopyRef)
157 #define ddot FORTRAN_WRAPPER(ddotRef)
158 #define dscal FORTRAN_WRAPPER(dscalRef)
159 #else
160 #if defined( _USE_32_BIT_BLAS )
161 #define daxpy FORTRAN_WRAPPER(daxpy32)
162 #define dcopy FORTRAN_WRAPPER(dcopy32)
163 #define ddot FORTRAN_WRAPPER(ddot32)
